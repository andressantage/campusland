# Programa para calcular la comisión de ventas de vendedores
# Autor: Juan Esteban Gómez García
# Fecha: 03/03/2023

cedula = int(input("Cédula de ciudadanía del vendedor: "))

valorTotalVentas = 0
valorTotalComisiones = 0
while cedula != -1:
    nombre = input("Nombre del vendedor: ")
    tipoVendedor = int(input("Tipo de vendedor (1=Puerta a puerta, 2=Telemercadeo, 3=Ejecutivo de ventas): "))
    valorVentas = float(input("Valor de ventas realizadas en el mes: "))
    if tipoVendedor == 1:
        comisiones = valorVentas*0.2
    elif tipoVendedor == 2:
        comisiones = valorVentas*0.15
    else:
        comisiones = valorVentas*0.25
    valorTotalComisiones = valorTotalComisiones + comisiones
    valorTotalVentas = valorTotalVentas + valorVentas
    print("Valor a pagar a", nombre, "es: ", "{:,.2f}".format(comisiones))
    cedula = int(input("Cédula de ciudadanía del vendedor: "))
print("Valor total de ventas es: ", "{:,.2f}".format(valorTotalVentas))
print("Valor total a pagar por comisiones es: ", "{:,.2f}".format(valorTotalComisiones))

        pagarPasajes = valorPasajes*0.3

        pagarEncomiendas = valorEncomiendas*0.15
        totalNovatos = totalNovatos + 1
    valorPagar = pagarPasajes + pagarEncomiendas
    totalPagar = totalPagar + valorPagar
    print("Nombre del conductor: ", nombre)
