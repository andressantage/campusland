    encomiendas=float(input("Valor total por concepto de encomiendas del mes: "))

    #c2 es el porcentaje de la comision para encomienda segun clase de conductor
    if clase==1:
        c1=0.3
        c2=0.2
        n_expertos=n_expertos+1
        
    else:
        c1=0.2
        c2=0.15
        n_novatos=n_novatos+1
    
    pago_pasajes=pasajes*c1
    pago_encomiendas=encomiendas*c2
    total=total+pago_encomiendas+pago_pasajes
    registro.append([cedula,nombre,clase,pasajes,encomiendas,pago_pasajes,pago_encomiendas])
    print("\n")

for item in registro:

    print("\n")
